1function res = esim(theta,u)
 2
 3lam=theta(1);
 4sig=theta(2);
 5xi =theta(3);
 6
 7nn=poissrnd(lam*100);
 8
 9z=unifrnd(-50,50,nn,1);
10
11a=(u+sig*((1-rand(nn,1)).^(-xi)-1)/xi)/2;
12
13b=rand(nn,1).*a;
14cc=rand(nn,1).*a;
15
16pars = [a b cc];
17for i = 1:nn
18    pars(i,:) = pars(i,randsample(1:3,3));
19end
20
21a=pars(:,1);
22b=pars(:,2);
23cc=pars(:,3);
24
25dm = max([a b cc],[],2);
26
27
28th=unifrnd(0,2*pi,nn,1);
29ph=unifrnd(0,2*pi,nn,1);
30p=-sin(ph)./z;
31q=cos(ph).*sin(th)./z;
32r=cos(ph).*cos(th)./z;
33
34
35den=r.^2.*cc.^2;

39D1=-p./den;
40E1=-q./den;
41F1=1./den-1;
42
43tt=atan(2*B1./(A1-C1))/2;
44%tt(isnan(tt)) = 0;
45
46A2=A1.*(cos(tt)).^2+2*B1.*sin(tt).*cos(tt)+C1.*(sin(tt)).^2;
47B2=A1.*(sin(tt)).^2-2*B1.*sin(tt).*cos(tt)+C1.*(cos(tt)).^2;
48D2=D1.*cos(tt)+E1.*sin(tt);
49E2=-D1.*sin(tt)+E1.*cos(tt);
50F2=F1;
51
52% A3=2*sqrt(D2.^2./A2.^2+E2.^2./(A2.*B2)-F2./A2);
53% B3=2*sqrt(D2.^2./(A2.*B2)+E2.^2./B2.^2-F2./B2);
54temp = D2.^2./A2.^2+E2.^2./(A2.*B2)-F2./A2;
55A3=2*sqrt(temp(temp >= 0));
56temp = D2.^2./(A2.*B2)+E2.^2./B2.^2-F2./B2;
57B3=2*sqrt(temp(temp >= 0));
58
59
60res = max([A3 B3],[],2);
61
62dm=dm(res>u);
63res = res(res>u);
64
65end
